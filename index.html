-- UPI Payment Risk Analysis
WITH monthly_risk AS (
  SELECT
    month, total_txn,
    failure_rate,
    ROUND((failure_rate -
    AVG(failure_rate)
    OVER()) / STDDEV(failure_rate)
    OVER(), 3) AS z_score
  FROM npci_upi_data
)
SELECT * FROM monthly_risk
WHERE ABS(z_score) > 2;
▶ 3 anomaly months detected
